Beware of "this"

var myButton = {
  element: document.getElementById(‘myButton’),
  value: ‘myValue’,

  init: function() {
    this.element.onclick = this.handleClick;
  },

  handleClick: function() {
    alert(this.value);
  }
};

myButton.handleClick(); // works, this == myButton
// but clicking on the button sets “this” to element (or window)

Using "bind"

var myButton = {
  element: document.getElementById(‘myButton’),
  value: ‘myValue’,

  init: function() {
    this.handleClick = this.handleClick.bind(this);
    this.element.onclick = this.handleClick;
  },

  handleClick: function() {
    alert(this.value);
  }
};

Trick for anonymous handlers

var myButton = {
  element: document.getElementById(‘myButton’),
  value: ‘myValue’,

  init: function() {
    var self = this;
    this.element.onclick = function() {
      alert(self.value);
    };
  }
};

Backbone: Lightweight MV(C)

var Todo = Backbone.Model.extend({
  defaults: {
    title: "empty todo..."
  }
});

var TodoView = Backbone.View.extend({
  initialize: function() {
    this.listenTo(this.model, "change", this.render);
  },
  render: function() {
    this.$el.html(this.template(this.model.toJSON()));
    return this;
  }
});
var myTodo = new Todo({ title: "Buy milk" });
var myTodoView = new TodoView({ model: myTodo });
$("#todolist").append(myTodoView.render().$el);
myTodo.set("title", "Buy lowfat milk"); // view will update
