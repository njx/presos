Function replacement

function myFunc() {
    console.log('old message');
}
function myFunc() {
    var func, funcDef = __funcReplacements['myFunc'];
    if (funcDef) {
        func = eval(funcDef);
    } else {
        func = function () {
            console.log('old message');
        }
    }
    return func.apply(this, arguments);
}
__funcReplacements['myFunc'] = 
    '(function () { console.log(\'new message\'); })';
